```c
printf("%d" , /* 10진수 정수*/);

printf("%f" , /* 실수값(소숫점 6번째 자리까지 출력) */);

printf("%.4f" , /* 실수값(소숫점 4번째 자리까지 출력) */); 

printf("%o" , /* 8진수 정수 */);

printf("%x" , /* 16진수 정수*/);

printf("%e" , /* 지수표현 */);

printf("%g" , /* 자동변환 */);

//-------------------------------------------------

// 최소한 5자리를 출력한다. 5자리 미만에서는 오른쪽 정렬된다 
printf("%5d",/* 10진수 정수*/);

// 최소한 5자리를 출력한다.+ 공백은 0으로 출력한다.
printf("%05d",/* 10진수 정수*/);

// 최소한 5자리를 출력한다. 왼쪽 정렬되어 출력된다.
printf("%-5d",/* 10진수 정수*/);

//-------------------------------------------------

printf("%c", /* 문자 */);
printf("%s", /* 문자열 */);
```
